25int sfcx_read_page(unsigned char *data, int address, int raw)
26{
27 int status;
28
29 sfcx_writereg(SFCX_STATUS, sfcx_readreg(SFCX_STATUS));
30
31 // Set flash address (logical)
32 //address &= 0x3fffe00; // Align to page
33 sfcx_writereg(SFCX_ADDRESS, address);
34
35 // Command the read
36 // Either a logical read (0x200 bytes, no meta data)
37 // or a Physical read (0x210 bytes with meta data)
38 sfcx_writereg(SFCX_COMMAND, raw ? PHY_PAGE_TO_BUF : LOG_PAGE_TO_BUF);
39
40 // Wait Busy
41 while ((status = sfcx_readreg(SFCX_STATUS)) & STATUS_BUSY);
42
43 if (!SFCX_SUCCESS(status))
44 {
45 if (status & STATUS_BB_ER)
46 printf(" ! SFCX: Bad block found at %08X\n", sfcx_address_to_block(address));
47 else if (status & STATUS_ECC_ER)
48 // printf(" ! SFCX: (Corrected) ECC error at address %08X: %08X\n", address, status);
49 status = status;
50 else if (!raw && (status & STATUS_ILL_LOG))
51 printf(" ! SFCX: Illegal logical block at %08X (status: %08X)\n", sfcx_address_to_block(address), status);
52 else
53 printf(" ! SFCX: Unknown error at address %08X: %08X. Please worry.\n", address, status);
54 }
55
56 // Set internal page buffer pointer to 0
57 sfcx_writereg(SFCX_ADDRESS, 0);
58
59 int i;
60 int page_sz = raw ? sfc.page_sz_phys : sfc.page_sz;
61
62 for (i = 0; i < page_sz ; i += 4)
63 {
64 // Transfer data from buffer to register
65 sfcx_writereg(SFCX_COMMAND, PAGE_BUF_TO_REG);
66
67 // Read out our data through the register
68 *(int*)(data + i) = __builtin_bswap32(sfcx_readreg(SFCX_DATA));
69 }
70
71 return status;
72}

645unsigned int sfcx_init(void)
646{
647 if ((xenon_get_PCIBridgeRevisionID() >= 0x70) && (sfcx_readreg(SFCX_PHISON) != 0)) {
648 printf(" ! SFCX: Unsupported Type - PHISON eMMC\n");
649 return 3;
650 }
651
652 unsigned int config = sfcx_readreg(SFCX_CONFIG);
653
654 if (sfc.initialized) return config;
655
656 sfc.initialized = 0;
657 sfc.meta_type = 0;
658 sfc.page_sz = 0x200;
659 sfc.meta_sz = 0x10;
660 sfc.page_sz_phys = sfc.page_sz + sfc.meta_sz;
661
662 //Turn off interrupts, turn off WP_EN, and set DMA pages to 0
663 sfcx_writereg(SFCX_CONFIG, config &~ (CONFIG_INT_EN|CONFIG_WP_EN|CONFIG_DMA_LEN));
664
